


### nginx部署qiankun项目

```
server {
        listen      9009;
        server_name  localhost;

        #charset koi8-r;

        #access_log  logs/host.access.log  main;
        #主应用(基座配置)
        location / {   
            root   html;
            index  index.html index.htm;
			try_files $uri $uri/ /index.html;
             }
        #子应用配置
		location /sub/sub-vue {
			    alias html/subapp/sub-vue/;
				index  index.html index.htm;
		        try_files $uri $uri/ index.html;
		    }
```
